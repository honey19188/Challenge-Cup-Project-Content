class WS2812:
    def __init__(self, pin, num):
        self.pin = Pin(pin, Pin.OUT)
        self.num = num
        self.buf = bytearray(num * 3)
        self.rmt = esp32.RMT(0, pin=pin, clock_div=8)
        self.T0H = 4; self.T0L = 8; self.T1H = 8; self.T1L = 4

    def __setitem__(self, index, color):
        r, g, b = color
        if 0 <= index < self.num:
            self.buf[index * 3] = g
            self.buf[index * 3 + 1] = r
            self.buf[index * 3 + 2] = b

    def write(self):
        self.pin.value(0)
        time.sleep_us(300)
        pulses = []
        for byte in self.buf:
            for i in range(7, -1, -1):
                bit = (byte >> i) & 1
                if bit:
                    pulses.extend((self.T1H, self.T1L))
                else:
                    pulses.extend((self.T0H, self.T0L))
        self.rmt.write_pulses(tuple(pulses), 1)
        time.sleep_us(50)

class SSD1306(framebuf.FrameBuffer):
    def __init__(self, width, height, i2c, addr=0x3C):
        self.width = width
        self.height = height
        self.i2c = i2c
        self.addr = addr
        self.temp = bytearray(2)
        self.buffer = bytearray(((height // 8) * width) + 1)
        self.buffer[0] = 0x40
        super().__init__(memoryview(self.buffer)[1:], width, height, framebuf.MONO_VLSB)
        self._init()

    def _write_cmd(self, cmd):
        self.temp[0] = 0x80
        self.temp[1] = cmd
        self.i2c.writeto(self.addr, self.temp)

    def _init(self):
        for cmd in (0xAE,0x20,0x00,0x40,0xA1,0xA8,self.height-1,0xC8,
                    0xD3,0x00,0xDA,0x12 if self.height==64 else 0x02,
                    0xD5,0x80,0xD9,0xF1,0xDB,0x30,0x81,0xFF,
                    0xA4,0xA6,0x8D,0x14,0xAF):
            self._write_cmd(cmd)

    def show(self):
        self.i2c.writeto(self.addr, self.buffer)

def motor_control(direction, steps=1):
    global motor_running, motor_total, motor_step_idx, motor_dir
    if steps <= 0: return
    motor_dir = 1 if direction > 0 else -1
    motor_total = steps
    motor_step_idx = 0
    motor_running = True

def motor_process():
    global motor_running, motor_step_idx, motor_total, motor_dir
    if not motor_running: return
    for _ in range(80):
        if motor_step_idx >= motor_total:
            for p in motor_pins: p.value(0)
            motor_running = False
            return
        idx = motor_step_idx % 8
        seq = FULL_STEPS[idx] if motor_dir > 0 else FULL_STEPS[7 - idx]
        motor_step(seq)
        time.sleep_us(motor_speed)
        motor_step_idx += 1
    time.sleep_ms(1)

def motor_rotate(degrees):
    steps = abs(degrees) * 4096 // 360
    motor_control(1 if degrees > 0 else -1, steps)

def hsv2rgb(h, s, v):
    if s == 0: return (v, v, v)
    region = h // 43
    rem = (h - region * 43) * 6
    p = (v * (255 - s)) // 255
    q = (v * (255 - (s * rem) // 255)) // 255
    t = (v * (255 - (s * (255 - rem)) // 255)) // 255
    if region == 0: return (v, t, p)
    if region == 1: return (q, v, p)
    if region == 2: return (p, v, t)
    if region == 3: return (p, q, v)
    if region == 4: return (t, p, v)
    return (v, p, q)
